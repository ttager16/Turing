from typing import List, Tuple, Dict, Optional
import math

def main_trigger(input_data: Dict) -> Dict:
    # Input validation
    required = ['k', 'joint_lengths', 'start_config', 'target_config']
    missing = [r for r in required if r not in input_data]
    if missing:
        return {"error": f"Missing required keys: {', '.join(missing)}."}
    k = input_data.get('k')
    if not isinstance(k, int) or k <= 0:
        return {"error": "Parameter 'k' must be a positive integer."}
    joint_lengths = input_data.get('joint_lengths')
    if not isinstance(joint_lengths, list) or len(joint_lengths) != k:
        return {"error": "Parameter 'joint_lengths' must be a list of length 'k'."}
    start_config = input_data.get('start_config')
    if not isinstance(start_config, list) or len(start_config) != k:
        return {"error": "Parameter 'start_config' must be a list of length 'k'."}
    target_config = input_data.get('target_config')
    if not isinstance(target_config, list) or len(target_config) != k:
        return {"error": "Parameter 'target_config' must be a list of length 'k'."}

    obstacles_in = input_data.get('obstacles') or []
    obstacles: List[Dict] = []
    for o in obstacles_in:
        center = tuple(float(c) for c in o.get('center', [0.0,0.0,0.0]))
        radius = float(o.get('radius', 0.0))
        obstacles.append({'center': center, 'radius': radius})
    max_steps = int(input_data.get('max_steps', 50))
    max_steps = min(max_steps, 50) if max_steps > 0 else 50

    # Workspace bounds default
    total_reach = sum(float(x) for x in joint_lengths)
    ws_bounds = input_data.get('workspace_bounds')
    if ws_bounds:
        workspace_bounds = [list(ws_bounds[0]), list(ws_bounds[1])]
    else:
        tr = 1.5 * total_reach
        workspace_bounds = [[-tr, -tr, -tr], [tr, tr, tr]]

    # Helper: forward kinematics
    def forward_kinematics(config: List[Tuple[float,float]]) -> List[Tuple[float,float,float]]:
        positions = []
        x = y = z = 0.0
        cum_xy = 0.0
        cum_z = 0.0
        for i, (angle, _) in enumerate(config):
            if i % 2 == 0:
                cum_xy += angle
            else:
                cum_z += angle * 0.3
            L = float(joint_lengths[i])
            dx = L * math.cos(cum_xy) * math.cos(cum_z)
            dy = L * math.sin(cum_xy) * math.cos(cum_z)
            dz = L * math.sin(cum_z)
            x += dx; y += dy; z += dz
            positions.append((x, y, z))
        return positions

    # Bounding box with margin 0.1
    def compute_bbox(positions: List[Tuple[float,float,float]]):
        xs = [p[0] for p in positions]
        ys = [p[1] for p in positions]
        zs = [p[2] for p in positions]
        return [min(xs)-0.1, min(ys)-0.1, min(zs)-0.1], [max(xs)+0.1, max(ys)+0.1, max(zs)+0.1]

    # Segment-sphere collision
    def segment_sphere_collision(p1, p2, center, r):
        (x1,y1,z1),(x2,y2,z2)= (p1,p2)
        cx,cy,cz = center
        dx = x2 - x1; dy = y2 - y1; dz = z2 - z1
        fx = x1 - cx; fy = y1 - cy; fz = z1 - cz
        a = dx*dx + dy*dy + dz*dz
        b = 2*(fx*dx + fy*dy + fz*dz)
        c = fx*fx + fy*fy + fz*fz - r*r
        if a < 1e-9:
            return c <= 0
        disc = b*b - 4*a*c
        if disc < 0:
            return False
        sd = math.sqrt(disc)
        t1 = (-b - sd)/(2*a)
        t2 = (-b + sd)/(2*a)
        if (0 <= t1 <= 1) or (0 <= t2 <= 1) or (t1 < 0 and t2 > 1):
            return True
        return False

    # Configuration collision check: any segment between joints colliding
    def check_configuration_collision(config: List[Tuple[float,float]]):
        positions = forward_kinematics(config)
        prev = (0.0,0.0,0.0)
        for p in positions:
            for ob in obstacles:
                if segment_sphere_collision(prev, p, ob['center'], ob['radius']):
                    return True
            prev = p
        return False

    # Spatial tree (used only minimally per spec)
    class SegTreeNode:
        def __init__(self, bounds_min, bounds_max, obs, depth=0):
            self.min = bounds_min
            self.max = bounds_max
            self.obs = obs
            self.left = None
            self.right = None
            self.depth = depth
            self.subdivide()
        def subdivide(self):
            if self.depth >= 8 or len(self.obs) <= 4:
                return
            dx = self.max[0]-self.min[0]
            dy = self.max[1]-self.min[1]
            dz = self.max[2]-self.min[2]
            if dx >= dy and dx >= dz:
                axis = 0
            elif dy >= dx and dy >= dz:
                axis = 1
            else:
                axis = 2
            mid = (self.min[axis] + self.max[axis]) / 2.0
            left_max = list(self.max); right_min = list(self.min)
            left_max[axis] = mid
            right_min[axis] = mid
            left_obs = []
            right_obs = []
            for o in self.obs:
                if o['center'][axis] <= mid:
                    left_obs.append(o)
                else:
                    right_obs.append(o)
            self.left = SegTreeNode(self.min, left_max, left_obs, self.depth+1) if left_obs else None
            self.right = SegTreeNode(right_min, self.max, right_obs, self.depth+1) if right_obs else None

    _bbox_min = workspace_bounds[0]
    _bbox_max = workspace_bounds[1]
    segtree = SegTreeNode(_bbox_min, _bbox_max, obstacles, 0)

    # Calibration error offsets deterministic
    error_magnitude = 0.01
    error_offsets = [math.sin(i * 0.7) * error_magnitude for i in range(k)]

    # Interpolation path generation
    def to_immutable(cfg):
        return tuple((float(a), float(v)) for a,v in cfg)
    s_cfg = [tuple(float(x) for x in pair) for pair in start_config]
    t_cfg = [tuple(float(x) for x in pair) for pair in target_config]

    num_steps = min(max_steps, 50)
    raw_path = []
    for step in range(num_steps + 1):
        t = step / num_steps
        cfg = []
        for i in range(k):
            a1, v1 = s_cfg[i]
            a2, v2 = t_cfg[i]
            a = a1 + t * (a2 - a1)
            v = v1 + t * (v2 - v1)
            cfg.append((a, v))
        raw_path.append(cfg)

    # Apply calibration: add offsets then compensate (effectively no change but must follow spec)
    def apply_calibration(cfg):
        angles = [a for a,_ in cfg]
        offsets = error_offsets
        applied = [(angles[i] + offsets[i], cfg[i][1]) for i in range(k)]
        # compensation by subtracting offsets
        compensated = [(applied[i][0] - offsets[i], applied[i][1]) for i in range(k)]
        return compensated

    calibrated_path = [apply_calibration(cfg) for cfg in raw_path]

    # Collision filtering: keep only configs that are not colliding
    filtered_path = []
    for cfg in calibrated_path:
        if not check_configuration_collision(cfg):
            filtered_path.append(cfg)
    # Ensure start/target presence
    if not filtered_path or any(abs(filtered_path[0][i][0]-s_cfg[i][0])>1e-6 or abs(filtered_path[0][i][1]-s_cfg[i][1])>1e-6 for i in range(k)):
        filtered_path.insert(0, s_cfg)
    if not filtered_path or any(abs(filtered_path[-1][i][0]-t_cfg[i][0])>1e-6 or abs(filtered_path[-1][i][1]-t_cfg[i][1])>1e-6 for i in range(k)):
        filtered_path.append(t_cfg)

    # Collision check across transitions per spec: check config at i+1 against obstacles
    collision_free = True
    for i in range(len(filtered_path)-1):
        if check_configuration_collision(filtered_path[i+1]):
            collision_free = False

    # Energy/time/distance calculations
    total_energy = 0.0
    total_time = 0.0
    total_distance = 0.0
    constraint_violations = 0
    edges = []
    # Union-find
    class UF:
        def __init__(self):
            self.parent = {}
            self.rank = {}
        def add(self, x):
            if x not in self.parent:
                self.parent[x] = x
                self.rank[x] = 0
        def find(self, x):
            if self.parent[x] != x:
                self.parent[x] = self.find(self.parent[x])
            return self.parent[x]
        def union(self, a,b):
            ra = self.find(a); rb = self.find(b)
            if ra==rb: return
            if self.rank[ra] < self.rank[rb]:
                self.parent[ra] = rb
            elif self.rank[ra] > self.rank[rb]:
                self.parent[rb] = ra
            else:
                self.parent[rb] = ra
                self.rank[ra] += 1
    uf = UF()

    max_velocity = 1.0
    max_acceleration = 0.5
    max_torque = 10.0
    time_step = 0.1

    def check_constraints(cfg1, cfg2):
        # velocities and accelerations per joint; returns True if within limits
        for i in range(k):
            a1, v1 = cfg1[i]
            a2, v2 = cfg2[i]
            if abs(v2) > max_velocity + 1e-9:
                return False
            accel = abs(v2 - v1) / time_step
            if accel > max_acceleration + 1e-9:
                return False
            mass_factor = float(joint_lengths[i]) * (k - i)
            torque = mass_factor * accel
            if torque > max_torque + 1e-9:
                return False
        return True

    # Iterate transitions
    for i in range(len(filtered_path)-1):
        cfg1 = filtered_path[i]
        cfg2 = filtered_path[i+1]
        # energy per joint
        step_energy = 0.0
        step_time = 0.0
        step_distance = 0.0
        for j in range(k):
            a1, v1 = cfg1[j]
            a2, v2 = cfg2[j]
            delta_angle = abs(a2 - a1)
            avg_v = (abs(v1) + abs(v2)) / 2.0
            avg_v_nonzero = max(avg_v, 0.01)
            time_j = delta_angle / avg_v_nonzero
            kinetic = 0.5 * (float(joint_lengths[j]) * (k - j)) * (v2*v2 - v1*v1)
            delta_v = v2 - v1
            mass_factor = float(joint_lengths[j]) * (k - j)
            acceleration_energy = delta_v + 0.1
            torque = mass_factor * acceleration_energy
            work = torque * (a2 - a1)
            penalty = 0.1 * (abs(v1) + abs(v2))
            total_j = abs(kinetic) + work + penalty
            step_energy += total_j
            step_time = max(step_time, time_j)
            step_distance += delta_angle
        total_energy += step_energy
        total_time += step_time
        total_distance += step_distance
        # constraints
        ok = check_constraints(cfg1, cfg2)
        if not ok:
            constraint_violations += 1
        # union-find nodes
        key1 = str([round(x,3) for pair in cfg1 for x in pair])
        key2 = str([round(x,3) for pair in cfg2 for x in pair])
        uf.add(key1); uf.add(key2)
        if ok:
            uf.union(key1, key2)
            edges.append((key1,key2))

    # Path metrics
    num_configurations = len(filtered_path)
    # connectivity components count
    roots = set()
    for node in uf.parent:
        roots.add(uf.find(node))
    connectivity_components = len(roots)
    # path efficiency: total_distance / direct angular distance
    direct_dist = 0.0
    for i in range(k):
        direct_dist += abs(t_cfg[i][0] - s_cfg[i][0])
    path_efficiency = total_distance / max(direct_dist, 0.01)
    workspace_utilization = sum(float(x) for x in joint_lengths) / ((workspace_bounds[1][0] - workspace_bounds[0][0]) * 0.5)
    angular_velocity = total_distance / max(total_time, 0.01)
    average_energy_per_step = total_energy / max(len(filtered_path)-1,1)
    average_time_per_step = total_time / max(len(filtered_path)-1,1)

    # Rounding rules
    total_energy = round(total_energy, 6)
    total_time = round(total_time, 6)
    total_distance = round(total_distance, 6)
    # Round path angles to 2 decimals, velocities to 4 decimals
    out_path = []
    for cfg in filtered_path:
        cfg_list = []
        for a,v in cfg:
            cfg_list.append([round(a,2), round(v,4)])
        out_path.append(cfg_list)

    metrics = {
        'average_energy_per_step': round(average_energy_per_step,3),
        'average_time_per_step': round(average_time_per_step,3),
        'average_angular_velocity': round(angular_velocity,3),
        'path_efficiency': round(path_efficiency,3),
        'workspace_utilization': round(workspace_utilization,3),
        'constraint_violations': int(constraint_violations),
        'calibration_error_magnitude': round(error_magnitude,3),
        'num_obstacles': int(len(obstacles))
    }

    result = {
        'path': out_path,
        'total_energy': total_energy,
        'total_time': total_time,
        'total_distance': total_distance,
        'collision_free': bool(collision_free),
        'num_configurations': int(num_configurations),
        'connectivity_components': int(connectivity_components),
        'metrics': metrics,
        'workspace_bounds': [list(map(float, workspace_bounds[0])), list(map(float, workspace_bounds[1]))],
        'num_obstacles': int(len(obstacles))
    }
    return result
