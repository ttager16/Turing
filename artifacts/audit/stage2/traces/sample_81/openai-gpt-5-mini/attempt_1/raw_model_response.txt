from typing import List, Any, Dict, Optional
from threading import RLock
from collections import defaultdict, deque
import bisect
import time

# Lightweight self-balancing like structure using sorted lists per ward keyed by (priority, emergency_flag, arrival_seq)
# Global registry maps id -> record
# Concurrency simulated via RLock for atomic operations
# Segment update optimization: when changing priority or transferring, only affected ward lists are updated
# Mass updates handled by batch operations with minimal scans using mapping by priority groups

class Patient:
    __slots__ = ('id', 'ward', 'priority', 'emergency', 'duration', 'seq')
    def __init__(self, pid:int, ward:str, priority:int, emergency:bool, duration:int, seq:int):
        self.id = pid
        self.ward = ward
        self.priority = priority
        self.emergency = emergency
        self.duration = duration
        self.seq = seq

    def key(self):
        # Lower priority value -> higher urgency; emergency first
        return (self.priority, 0 if self.emergency else 1, self.seq, self.id)

class WardQueue:
    def __init__(self):
        self.list = []  # sorted list of keys
        self.map = {}   # id -> Patient

    def insert(self, patient: Patient):
        k = patient.key()
        bisect.insort_left(self.list, (k, patient.id))
        self.map[patient.id] = patient

    def remove(self, pid: int) -> Optional[Patient]:
        p = self.map.pop(pid, None)
        if p is None:
            return None
        k = p.key()
        i = bisect.bisect_left(self.list, (k, pid))
        # scan around to find exact id
        n = len(self.list)
        for j in range(max(0, i-2), min(n, i+3)):
            if self.list[j][1] == pid:
                self.list.pop(j)
                break
        return p

    def update_priority(self, pid:int, new_priority:int):
        p = self.map.get(pid)
        if not p:
            return None
        # remove and reinsert
        self.remove(pid)
        p.priority = new_priority
        self.insert(p)
        return p

    def transfer_out(self, pid:int):
        return self.remove(pid)

    def quick_bulk_repriority(self, ids, new_priority):
        # Only process provided ids; remove and reinsert
        updated = []
        for pid in ids:
            if pid in self.map:
                p = self.map[pid]
                self.remove(pid)
                p.priority = new_priority
                self.insert(p)
                updated.append(pid)
        return updated

class Hospital:
    def __init__(self):
        self.lock = RLock()
        self.wards: Dict[str, WardQueue] = defaultdict(WardQueue)
        self.registry: Dict[int, Patient] = {}
        self.seq_counter = 0
        # For quick lookup per priority groups across wards for mass ops
        self.priority_index: Dict[int, set] = defaultdict(set)

    def _next_seq(self):
        self.seq_counter += 1
        return self.seq_counter

    def insert(self, info: Dict[str,Any]) -> str:
        with self.lock:
            pid = info['id']
            if pid in self.registry:
                return f"Patient {pid} already exists"
            ward = info['ward']
            priority = info.get('priority', 999)
            emergency = bool(info.get('emergency', False))
            duration = info.get('duration', 0)
            seq = self._next_seq()
            p = Patient(pid, ward, priority, emergency, duration, seq)
            self.registry[pid] = p
            self.wards[ward].insert(p)
            self.priority_index[priority].add(pid)
            return f"Patient {pid} added to ward {ward}"

    def delete(self, info: Dict[str,Any]) -> str:
        with self.lock:
            pid = info['id']
            p = self.registry.pop(pid, None)
            if not p:
                return f"Patient {pid} not found"
            self.wards[p.ward].remove(pid)
            # remove from priority index
            if p.priority in self.priority_index and pid in self.priority_index[p.priority]:
                self.priority_index[p.priority].remove(pid)
            return f"Patient {pid} removed from system"

    def transfer(self, info: Dict[str,Any]) -> str:
        with self.lock:
            pid = info['id']
            new_ward = info['new_ward']
            p = self.registry.get(pid)
            if not p:
                return f"Patient {pid} not found"
            if p.ward == new_ward:
                return f"Patient {pid} already in ward {new_ward}"
            # remove from old ward and insert into new ward, preserving seq and priority
            old_ward = p.ward
            self.wards[old_ward].transfer_out(pid)
            p.ward = new_ward
            self.wards[new_ward].insert(p)
            return f"Patient {pid} transferred to ward {new_ward}"

    def reorder(self, info: Dict[str,Any]) -> str:
        with self.lock:
            pid = info['id']
            new_priority = info['new_priority']
            simulate_mass = bool(info.get('simulate_mass_update', False))
            p = self.registry.get(pid)
            if not p:
                return f"Patient {pid} not found"
            old_priority = p.priority
            if simulate_mass:
                # Simulate mass reprioritization: all patients in same old priority across hospital get moved
                affected = list(self.priority_index.get(old_priority, set()))
                # Update priority index efficiently
                if affected:
                    for apid in affected:
                        ap = self.registry.get(apid)
                        if not ap:
                            continue
                        # remove from old ward list and update
                        self.wards[ap.ward].remove(apid)
                        ap.priority = new_priority
                        self.wards[ap.ward].insert(ap)
                    # move index set
                    self.priority_index[new_priority].update(affected)
                    self.priority_index[old_priority].clear()
                return f"Patient {pid} updated to priority {new_priority} with mass update simulation"
            else:
                # single update
                self.wards[p.ward].update_priority(pid, new_priority)
                # update registry and index
                if pid in self.priority_index.get(old_priority, set()):
                    self.priority_index[old_priority].remove(pid)
                self.priority_index[new_priority].add(pid)
                return f"Patient {pid} updated to priority {new_priority}"

    def process_command(self, cmd):
        op = cmd[0]
        info = cmd[1] if len(cmd) > 1 else {}
        if op == 'insert':
            return self.insert(info)
        elif op == 'delete':
            return self.delete(info)
        elif op == 'transfer':
            return self.transfer(info)
        elif op == 'reorder':
            return self.reorder(info)
        else:
            return f"Unknown command {op}"

# Entry function
def manage_patient_queue(commands: List[List]) -> List[str]:
    hosp = Hospital()
    outputs = []
    start_time = time.time()
    for cmd in commands:
        try:
            res = hosp.process_command(cmd)
        except Exception as e:
            res = f"Error processing command {cmd[0]}: {e}"
        outputs.append(res)
        # Ensure responsiveness: if many commands, we could yield internally; here we just track time
        if (time.time() - start_time) > 0.49:
            # simulate propagation deadline: quick sync checkpoint
            with hosp.lock:
                pass
            start_time = time.time()
    return outputs
