from typing import List, Dict, Any

def optimize_traffic_flow(sensor_data: List[Dict[str, Any]]) -> Dict[str, Any]:
    # Prepare output structures
    signal_adjustments = []
    alternate_routes = []

    if not sensor_data:
        return {"signal_adjustments": signal_adjustments, "alternate_routes": alternate_routes}

    # Maintain order as given (sequential network)
    ids = [entry.get("intersection_id") for entry in sensor_data]

    # Compute signal adjustments using provided proportional control formula
    for entry in sensor_data:
        intersection_id = entry.get("intersection_id")
        vehicle_count = float(entry.get("vehicle_count", 0))
        average_speed = float(entry.get("average_speed", 0))
        congestion_level = float(entry.get("congestion_level", 0))

        raw = 1 - (0.5 * congestion_level + 0.3 * (vehicle_count / 300.0) - 0.2 * (average_speed / 60.0))
        signal_adjustment = max(0.1, min(1.0, raw))
        # Round to 2 decimal places as in sample (but keep float)
        signal_adjustments.append({
            "intersection_id": intersection_id,
            "signal_adjustment": round(signal_adjustment, 2)
        })

    # Alternate route generation
    n = len(ids)
    if n < 3:
        return {"signal_adjustments": signal_adjustments, "alternate_routes": alternate_routes}

    # Find most congested intersection (highest congestion_level). If tie, pick first occurrence.
    max_idx = 0
    max_cong = float(sensor_data[0].get("congestion_level", 0))
    for i, entry in enumerate(sensor_data):
        c = float(entry.get("congestion_level", 0))
        if c > max_cong:
            max_cong = c
            max_idx = i

    # For each other intersection, suggest alternate route bypassing the most congested one:
    # Alternate route should connect to the next two consecutive intersections ahead in the sequence, if available.
    # That is, from current index i, identify a path that skips the congested node and goes to i+3 via i+1 and i+2,
    # but only if those nodes exist and the path bypasses the congested node.
    for i, origin in enumerate(ids):
        if i == max_idx:
            continue
        # We need next two nodes ahead from origin (i+1, i+2). Destination is i+2 (the second ahead).
        next1_idx = i + 1
        next2_idx = i + 2
        if next2_idx >= n:
            continue  # not enough following nodes
        # Ensure the alternate route bypasses the most congested node: the congested node index must be among the skipped ones.
        # The standard interpretation in examples: bypass the most congested intersection, so the route should skip it.
        # If the congested node is not between origin and destination along the normal path, still produce route that skips it.
        # To follow requirements: generate "From origin to destination via middle" where middle is next1, destination is next2,
        # but only if by skipping the congested node (i.e., the congested node equals next1 or next2) OR if skipping is meaningful.
        # We'll generate routes that skip the congested node if the congested node index is between origin and destination inclusive of next1/next2.
        # Specifically, if max_idx == next1_idx or max_idx == next2_idx:
        # Additionally, sample shows for A1 when B2 is max (B2 at index1), A1->D4 via C3 (A1 index0 -> C3 index2 via index1) which matches.
        if max_idx in (next1_idx, next2_idx):
            origin_id = origin
            via_id = ids[next1_idx]
            dest_id = ids[next2_idx]
            alternate_routes.append(f"Alternate Route: From {origin_id} to {dest_id} via {via_id}")

    return {"signal_adjustments": signal_adjustments, "alternate_routes": alternate_routes}
