import math

def _validate_inputs(n, num_layers, risk_threshold, concurrent_updates):
    if not (isinstance(n, int) and n >= 0):
        return False, "n must be a non-negative integer"
    if not (isinstance(num_layers, int) and num_layers > 0):
        return False, "num_layers must be a positive integer"
    if not (isinstance(risk_threshold, (int, float)) and risk_threshold >= 0):
        return False, "risk_threshold must be a non-negative number"
    if not (isinstance(concurrent_updates, int) and concurrent_updates >= 0):
        return False, "concurrent_updates must be a non-negative integer"
    return True, ""

def _catalan_sequence_dp(n):
    # DP per constraints: array size n+1, first two elements 1
    seq = [1] * (n + 1 if n >= 1 else 1)
    if n == 0:
        return [1]
    seq[0] = 1
    seq[1] = 1
    for i in range(2, n + 1):
        total = 0
        for j in range(i):
            total += seq[j] * seq[i - 1 - j]
        seq[i] = total
    return seq

def _bell_number(n):
    # Bell triangle method per constraints
    if n == 0:
        return 1
    bell = [[1]]
    for i in range(1, n + 1):
        row = [bell[i - 1][-1]]
        for j in range(1, i + 1):
            row.append(row[-1] + bell[i - 1][j - 1])
        bell.append(row)
    return bell[n][0]

def count_non_crossing_partitions_analysis(n: int, num_layers: int = 3, 
                                           risk_threshold: float = 0.5,
                                           concurrent_updates: int = 10) -> dict:
    valid, err = _validate_inputs(n, num_layers, risk_threshold, concurrent_updates)
    input_params = {
        "n": n,
        "num_layers": num_layers,
        "risk_threshold": risk_threshold,
        "concurrent_updates": concurrent_updates
    }
    if not valid:
        return {
            "success": False,
            "error": err,
            "input_parameters": input_params
        }
    # Catalan sequence
    catalan_seq = _catalan_sequence_dp(n)
    primary_count = catalan_seq[-1]
    # Layer analysis
    layer_partitions = []
    if n == 0:
        base_alloc = 0
        remainder = 0
        avg_nodes = 0.0
        # each layer has 0 nodes and 1 partition
        for i in range(num_layers):
            layer_partitions.append({
                "layer_index": i,
                "nodes": 0,
                "partitions": 1
            })
        total_distributions = 1
        avg_nodes_rounded = round(avg_nodes, 3)
        layer_valid = True
    else:
        base_alloc = n // num_layers
        remainder = n % num_layers
        nodes_per_layer = []
        for i in range(num_layers):
            nodes = base_alloc + (1 if i < remainder else 0)
            nodes_per_layer.append(nodes)
            # partitions for layer: Catalan of nodes; ensure sequence long enough
            part = _catalan_sequence_dp(nodes)[-1]
            layer_partitions.append({
                "layer_index": i,
                "nodes": nodes,
                "partitions": part
            })
        total_distributions = 1
        for lp in layer_partitions:
            total_distributions *= lp["partitions"]
        avg_nodes = sum(nodes_per_layer) / num_layers
        avg_nodes_rounded = round(avg_nodes, 3)
        layer_valid = True
    # Growth analysis
    growth = {}
    growth["current_value"] = primary_count
    if n < 2:
        growth["growth_ratio"] = None
        growth["growth_trend"] = "insufficient_data"
    else:
        prev = catalan_seq[-2]
        growth["previous_value"] = prev
        if prev <= 0:
            growth_ratio = None
        else:
            growth_ratio = round(primary_count / prev, 5)
        growth["growth_ratio"] = growth_ratio
        if growth_ratio is None:
            trend = "insufficient_data"
        else:
            if growth_ratio < 2:
                trend = "subexponential"
            elif growth_ratio >= 3.5:
                trend = "approaching_asymptotic_limit"
            else:
                trend = "exponential"
        growth["growth_trend"] = trend
    # Risk assessment
    total_partitions = primary_count
    diversity = math.log(total_partitions + 1) / 23.0
    if diversity > 1.0:
        diversity = 1.0
    diversity = round(diversity, 6)
    risk_score = round(1.0 - diversity, 4)
    thr_rounded = round(risk_threshold, 2)
    if risk_score < (0.5 * risk_threshold):
        risk_level = "low"
    elif risk_score < risk_threshold:
        risk_level = "moderate"
    elif risk_score < (1.5 * risk_threshold):
        risk_level = "high"
    else:
        risk_level = "critical"
    # Structural metrics
    if n == 0:
        avg_partition_size = 0.0
    else:
        avg_partition_size = (n + 1) / 2.0
    avg_partition_size = round(avg_partition_size, 1)
    if n >= 2:
        first_derivative = catalan_seq[-1] - catalan_seq[-2]
    else:
        first_derivative = None
    if n >= 3:
        second_derivative = (catalan_seq[-1] - catalan_seq[-2]) - (catalan_seq[-2] - catalan_seq[-3])
    else:
        second_derivative = None
    # Complexity metrics
    if n == 0:
        base_complexity = 1.0
    else:
        base_complexity = n * math.log(n + 1)
    base_complexity = round(base_complexity, 8)
    total_complexity = round(base_complexity * concurrent_updates, 9)
    estimated_throughput = None
    if total_complexity > 0:
        estimated_throughput = round(1000000.0 / total_complexity, 10)
    # Bell comparison
    bell_comp = None
    if 0 <= n <= 20:
        bell = _bell_number(n)
        non_crossing_ratio = round(primary_count / bell, 7) if bell != 0 else 0.0
        crossing = bell - primary_count
        bell_comp = {
            "non_crossing_partitions": primary_count,
            "total_partitions": bell,
            "non_crossing_ratio": non_crossing_ratio,
            "crossing_partitions": crossing
        }
    # Assemble result
    result = {
        "success": True,
        "input_parameters": input_params,
        "primary_result": {
            "non_crossing_partitions": primary_count,
            "description": f"Number of valid non-crossing partitions for {n} nodes"
        },
        "catalan_sequence": {
            "sequence": catalan_seq,
            "length": len(catalan_seq),
            "description": "Sequence of Catalan numbers from C_0 to C_n"
        },
        "layer_analysis": {
            "valid": layer_valid,
            "total_distributions": total_distributions,
            "average_nodes_per_layer": avg_nodes_rounded,
            "layer_partitions": layer_partitions
        },
        "growth_analysis": growth,
        "risk_assessment": {
            "total_partitions": total_partitions,
            "diversity_score": diversity,
            "risk_score": risk_score,
            "risk_level": risk_level,
            "threshold": thr_rounded
        },
        "structural_metrics": {
            "total_nodes": n,
            "total_partitions": total_partitions,
            "average_partition_size": avg_partition_size,
            "first_derivative": first_derivative,
            "second_derivative": second_derivative,
            "complexity_class": "exponential"
        },
        "concurrent_analysis": {
            "nodes": n,
            "concurrent_updates": concurrent_updates,
            "base_complexity": base_complexity,
            "total_complexity": total_complexity,
            "estimated_throughput": estimated_throughput
        }
    }
    if bell_comp is not None:
        result["bell_comparison"] = bell_comp
    return result
