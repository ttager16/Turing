from typing import List, Dict, Any, Set, Tuple

def allocate_minimum_teams(projects: List[Dict[str, Any]], teams: List[Dict[str, Any]]) -> int:
    # Flatten all phases into tasks with required_skills and deadline
    tasks: List[Tuple[Set[str], int]] = []
    for proj in projects:
        for ph in proj.get('phases', []):
            req = set(ph.get('required_skills', []))
            dl = ph.get('deadline', proj.get('project_deadline', 0))
            tasks.append((req, dl))
    # Sort tasks by deadline (earliest first) for greedy scheduling
    tasks.sort(key=lambda x: x[1])

    # Precompute team skill sets and availability
    team_skills: List[Set[str]] = []
    team_avail: List[int] = []
    for t in teams:
        skills = set()
        for m in t.get('members', []):
            skills.update(m.get('skills', []))
        team_skills.append(skills)
        team_avail.append(int(t.get('base_availability', 0)))

    n_teams = len(teams)
    if not tasks:
        return 0

    # Quick impossible check: union of all skills must cover each task
    all_skills = set().union(*team_skills) if team_skills else set()
    for req, _ in tasks:
        if not req.issubset(all_skills):
            return -1  # impossible to cover some task

    # Feasibility: given k teams (choose first k teams after sorting by something), check if they can schedule tasks.
    # We will choose the k most "capable" teams by total skills size and availability as heuristic.
    team_indices_sorted = list(range(n_teams))
    team_indices_sorted.sort(key=lambda i: ( -len(team_skills[i]), -team_avail[i] ))

    def feasible(k: int) -> bool:
        if k == 0:
            return False
        sel_indices = team_indices_sorted[:k]
        # For each selected team, track remaining availability (days)
        rem_avail = {i: team_avail[i] for i in sel_indices}
        # For scheduling, for each task we need to assign enough teams whose combined skills cover req.
        # We will greedily assign minimal number of teams: pick teams that add most uncovered skills per day cost.
        for req, dl in tasks:
            # we must allocate some total days of work not exceeding each team's remaining availability and before deadline.
            # Simplify: assume each phase requires 1 day of work total (not specified), but deadline constraints relative to availability:
            # We interpret that a team must have availability >=1 and deadline must be >=1. To incorporate deadlines, ensure sum of availabilities of selected teams that can contribute before dl >=1.
            # To be more general, assume each phase needs 1 unit of effort.
            needed_skills = set(req)
            # Choose subset of teams that together cover needed_skills and have at least 1 day remaining and rem_avail <= dl (we only ensure availability positive)
            candidates = [i for i in sel_indices if rem_avail[i] > 0 and team_skills[i] & needed_skills]
            # Greedy cover by selecting team that covers most uncovered skills
            used = []
            while needed_skills and candidates:
                best = max(candidates, key=lambda i: len(team_skills[i] & needed_skills))
                cover = team_skills[best] & needed_skills
                if not cover:
                    break
                needed_skills -= cover
                used.append(best)
                candidates.remove(best)
                # also remove any candidate with no overlap remaining
                candidates = [i for i in candidates if team_skills[i] & needed_skills]
            if needed_skills:
                return False
            # Now deduct 1 day of availability from one of the used teams (or distribute). We deduct from team among used with max remaining.
            # If no used (shouldn't happen), fail
            if not used:
                return False
            best_team = max(used, key=lambda i: rem_avail[i])
            rem_avail[best_team] -= 1
            if rem_avail[best_team] < 0:
                return False
        return True

    lo, hi = 1, n_teams
    ans = -1
    while lo <= hi:
        mid = (lo + hi) // 2
        if feasible(mid):
            ans = mid
            hi = mid - 1
        else:
            lo = mid + 1
    return ans if ans != -1 else -1
