from typing import List
import bisect

def manage_high_freq_trades(
    markets: List[List[float]],
    operations: List[str],
    prices: List[float],
    market_indices: List[int]
) -> List[int]:
    # Maintain each market as a sorted list; ensure input markets are copied/sorted
    M = [sorted(m) for m in markets]
    results: List[int] = []
    n = len(operations)
    for i in range(n):
        op = operations[i]
        price = prices[i]
        midx = market_indices[i]
        if midx < 0 or midx >= len(M):
            # invalid market index: FIND returns -1, UPDATE ignored
            if op == "FIND":
                results.append(-1)
            continue
        market = M[midx]
        if op == "FIND":
            # binary search for exact match
            lo = bisect.bisect_left(market, price)
            if lo < len(market) and market[lo] == price:
                results.append(lo)
            else:
                results.append(-1)
        elif op == "UPDATE":
            # insert price preserving sort; if exact value exists, replace the first occurrence
            lo = bisect.bisect_left(market, price)
            if lo < len(market) and market[lo] == price:
                market[lo] = price  # replace (idempotent for floats)
            else:
                market.insert(lo, price)
        else:
            # Unknown operation: ignore, but for FIND behave as not found
            if op == "FIND":
                results.append(-1)
    return results
